\section{Docker}
 
\defverbatim[colored]\lstupdatenodejs{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]
$ curl -sL https://deb.nodesource.com/setup_8.x | sudo -E bash - 
$ sudo apt-get update
$ sudo apt-get install -y nodejs
 \end{lstlisting}
}



\defverbatim[colored]\lstinstallyarn{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]
$ curl -sS https://dl.yarnpkg.com/debian/pubkey.gpg | sudo apt-key add -
$ echo "deb https://dl.yarnpkg.com/debian/ stable main" | sudo tee 
        /etc/apt/sources.list.d/yarn.list
$ sudo apt-get update && sudo apt-get install yarn
 \end{lstlisting}
}

\defverbatim[colored]\lstpokemon{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]

 $ git clone https://github.com/RN3r1/nuxtPokemonExample.git
 $ cd nuxtPokemonExample
 # install dependencies
 $ yarn install

 # serve with hot reload at localhost:3000
 $ yarn run dev

 # build for production and launch server
 $ yarn run build
 $ yarn start

 # generate static project
 $ yarn run generate
\end{lstlisting}
}


\defverbatim[colored]\lstpokemondocker{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]
$ git clone https://github.com/RN3r1/DockerNuxtPokemonExample.git
# install dependencies
$ yarn install

# serve with hot reload at localhost:3000
$ yarn run dev

# build for production and launch server
$ yarn run build
$ yarn start

# generate static project
$ yarn run generate

# generate Docker image
$ sudo docker build . -t <USER_HUB_DOCKER>/pokemon

# run Docker image
$ sudo docker run -p 80:3000 --name pokemon-container <USER_HUB_DOCKER>/pokemon
\end{lstlisting}
}

\defverbatim[colored]\lstpokemonhub{
 \begin{lstlisting}[language=python,showstringspaces=false, basicstyle={\tiny}, keywordstyle=\color{red}]
# start container
$ sudo docker start pokemon-container

# list containers
$ sudo docker container ls -a

# inspect the container
$ sudo docker inspect pokemon-container

# connect to container
$ sudo docker exec -it pokemon-container bash

# stop the container
$ sudo docker stop pokemon-container

$ sudo docker login

# push your image to Docker Cloud
$ sudo docker push <USER_HUB_DOCKER>/pokemon

# remove container
$ sudo docker rm <USER_HUB_DOCKER>/pokemon
\end{lstlisting}
}
